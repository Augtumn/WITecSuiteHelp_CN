C:\Program Data\WITec\WITec Suite 6.X\Common Files\WITec.SDK


使用 SDK 的示例代码：

using (WITecSDK witecSDK = new WITecSDK())

{

witecSDK.ConnectWithWriteAccess();

using (VideoImageAcquisition videoImageAcquisition = witecSDK.CreateVideoImageAcquisition())

{

Bitmap videoImage = videoImageAcquisition.AcquireVideoImage().GetAwaiter().GetResult();

}

}


不使用 SDK 的示例代码：

// 前提条件：添加对 "BasicUniversalCOMServerLib" 的引用（引用 -> 添加 -> COM -> BasicUniversalCOMServerLib）

static void Main(string[] args)

{

// 连接到 WITec Control

string serverName = Environment.MachineName;

Guid bucsCoreGuid = new Guid("C45E77CE-3D66-489A-B5E2-159F443BD1AA");

Type bucsCoreType = Type.GetTypeFromCLSID(bucsCoreGuid, serverName, true);

IBUCSCore wcCoreInterface = Activator.CreateInstance(bucsCoreType) as IBUCSCore;

// 获取参数修改器

IBUCSFloat parameterModifier = wcCoreInterface.GetSubSystemDefaultInterface("UserParameters|SequencerAsTimeGoesBy|IntegrationTime");

// 读取浮点参数的值

float value = parameterModifier.GetValue();

Console.WriteLine("参数浮点值: " + value);

// 请求写访问

// 前提条件：WITec Control 必须允许远程写访问（控制窗口，参数：COM 自动化 -> 允许远程访问）

((IBUCSAccess)wcCoreInterface).RequestWriteAccess(true);

// 写入浮点参数的值

parameterModifier.SetValue(0.42f);

// 获取所有可用子系统的列表

CBUCSSubSystemsList subSystemNameList = wcCoreInterface.GetSubSystemsList(null, 0);

for (int i = 0; i < subSystemNameList.GetNumberOfSystems(); i++)

{

Console.WriteLine("子系统名称: " + subSystemNameList.GetSubSystemName(i));

var subSystemName = subSystemNameList.GetSubSystemName(i);

var subSystemParameterList = wcCoreInterface.GetSubSystemsList(subSystemName, 10);

for (int j = 0; j < subSystemParameterList.GetNumberOfSystems(); j++)

{

Console.WriteLine("参数名称: " + subSystemParameterList.GetSubSystemName(j));

}

}

}
